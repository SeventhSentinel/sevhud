AddOptionMenu "OptionsMenu"
{
	StaticText "$OPTMNU_SEVHUD_VER"
	Submenu "$OPTMNU_SEVHUD", "SevHUDOptionMenu"
	StaticText ""
}

AddOptionMenu "OptionsMenuSimple"
{
	StaticText "$OPTMNU_SEVHUD_VER"
	Submenu "$OPTMNU_SEVHUD", "SEVHUDOptionMenu"
	StaticText ""
}

OptionValue "SevHUD_Counts"
{
	0, "$OPTMNU_SEVHUD_EXTRA_PERCENTAGES"
	1, "$OPTMNU_SEVHUD_EXTRA_COUNTS"
}

OptionValue "SevHUD_HideStealthHPBars"
{
	0, "$OPTVAL_NEVER"
	1, "$OPTMNU_SEVHUD_HP_BAR_HIDE_STEALTH_SOMETIMES"
	2, "$OPTVAL_ALWAYS"
}


OptionMenu "SevHUDOptionMenu"
{
	Title "$OPTMNU_SEVHUD"
	StaticText "$OPTMNU_USE_SEVHUD"
	Option "$ALTHUDMNU_ENABLE",					"hud_althud", "OnOff"
	StaticText ""
	IfGame(chex)
	{
		StaticText "$OPTMNU_SEVHUD_CHEXEXTRA_DESC"
		Option "$OPTMNU_SEVHUD_CHEXEXTRA",			"sevhud_chexextra", 			"OnOff"
		StaticText ""
	}
	StaticText "$OPTMNU_SEVHUD_AUTOMAP_DESC"
	Submenu "$OPTMNU_AUTOMAP",			"AutomapOptions"
	StaticText ""
	StaticText "$OPTMNU_SEVHUD_CROSSHAIR_DESC"
	Option "$HUDMNU_CROSSHAIR",				"crosshair", "Crosshairs"
	StaticText ""
	StaticText "$OPTMNU_SEVHUD_EXTRA_DESC"
	Option "$OPTMNU_SEVHUD_EXTRA", "sevhud_extra", "OnOff"
	Option "$OPTMNU_SEVHUD_EXTRA_STYLE", "sevhud_extra_counts", "SevHUD_Counts"
	StaticText ""
	Option "$OPTMNU_SEVHUD_HP_BAR", "sevhud_hpbar", "OnOff"
	StaticText "$OPTMNU_SEVHUD_HP_BAR_MIN_DESC_A"
	StaticText "$OPTMNU_SEVHUD_HP_BAR_MIN_DESC_B"
	TextField "$OPTMNU_SEVHUD_HP_BAR_MIN", "sevhud_hpbar_min", "sevhud_hpbar", 0
	Option "$OPTMNU_SEVHUD_HP_BAR_HIDE_STEALTH", "sevhud_hpbar_hidestealth", "SevHUD_HideStealthHPBars"
	StaticText ""
	StaticText "$OPTMNU_SEVHUD_UPDATE"
	StaticText ""
	StaticText "$OPTMNU_SEVHUD_SHILL", "orange"
}
